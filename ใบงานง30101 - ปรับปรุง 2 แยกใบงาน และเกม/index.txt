        window.onload = function() {
            const savedName = localStorage.getItem('innovatorName');
            if (savedName) {
                document.getElementById('studentNameInput').value = savedName;
            }
        }

        function registerAndGo() {
            const nameInput = document.getElementById('studentNameInput');
            const name = nameInput.value.trim();
            
            if (!name) {
                alert("กรุณากรอกชื่อ-สกุล ก่อนเข้าสู่ห้องเรียน");
                nameInput.focus();
                // Add shake animation or red border
                nameInput.classList.add('border-red-500', 'ring-2', 'ring-red-200');
                return;
            }
            
            localStorage.setItem('innovatorName', name);
            // เปลี่ยนเป็นชื่อไฟล์หน้าห้องเรียนของคุณ (เช่น classroom.html หรือ index.html)

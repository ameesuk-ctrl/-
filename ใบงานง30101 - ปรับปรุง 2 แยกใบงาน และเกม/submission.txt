        function switchTab(tab) {
            const formDiv = document.getElementById('content-form');
            const checkDiv = document.getElementById('content-check');
            const btnForm = document.getElementById('tab-btn-form');
            const btnCheck = document.getElementById('tab-btn-check');

            if (tab === 'form') {
                formDiv.classList.remove('hidden'); checkDiv.classList.add('hidden');
                btnForm.className = 'tab-active flex-1 py-3 text-sm transition-all';
                btnCheck.className = 'tab-inactive flex-1 py-3 text-sm transition-all';
            } else {
                checkDiv.classList.remove('hidden'); formDiv.classList.add('hidden');
                btnCheck.className = 'tab-active flex-1 py-3 text-sm transition-all';
                btnForm.className = 'tab-inactive flex-1 py-3 text-sm transition-all';
            }
            if(window.innerWidth < 1024) toggleSidebar();
        }

        function toggleSidebar() {
            const sidebar = document.getElementById('sidebar');
            const overlay = document.getElementById('sidebar-overlay');
            if (sidebar.classList.contains('-translate-x-full')) {
                sidebar.classList.remove('-translate-x-full'); overlay.classList.remove('hidden');
            } else {
                sidebar.classList.add('-translate-x-full'); overlay.classList.add('hidden');
            }
        }
